广义表

广义表中元素可以是单元素也可以是另一个广义表

  • 结构实现

    typedef struct GNode* GList;

struct GNode {
int Tag; /*标志域:0表示结点是单元素,1表示结点是广义表*/
union { /*子表指针域SubList与单元素数据域Dota复用,即共用储存空间*/
ElementType Dota;
GList SubList;
} URegion;
GList Next; /*指向后继结点*/
};

多重链表

  • 十字链表
    表示稀疏矩阵

    • 结构实现

      typedef struct ONode *OList;

struct Term {
int Tag; /*标志域:0表示结点是头结点,1表示结点是普通结点*/
int Row; /*头结点表示总行数,普通结点表示当前行数*/
int Col; /*同上*/
union {
ElementType Value;
int numbers;
} URegion;
};

struct ONode {
int Tag; /*标志域:0表示结点是Head,1表示结点是Term*/
union {
struct Term term;
OList Next; /*指向后继结点*/
} URegion;
OList Down, Right;
};
