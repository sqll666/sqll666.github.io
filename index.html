    // 比如这个
import java.util.Scanner;

public class Fraction {
    int numerator; // 分子
    int denominator; // 分母

Fraction(int numerator, int denominator)
{
    int gcd = 1;
    // 找到最大公约数(化简分数)
    for (int i = 2; i <= numerator && i <= denominator; i++) {
    if (numerator % i == 0 && denominator % i == 0) {
    gcd = i;
        }
    }
this.numerator = numerator / gcd;
this.denominator = denominator / gcd;
}

double toDouble()
{
return (double)numerator/denominator;
}

Fraction plus(Fraction in)
{
int denominator = this.denominator * in.denominator;
int numerator = this.numerator * in.denominator + in.numerator * this.denominator;
Fraction fraction = new Fraction(numerator, denominator);
return fraction;
}

Fraction multiply(Fraction in)
{
int denominator = this.denominator * in.denominator;
int numerator = this.numerator * in.numerator;
Fraction fraction = new Fraction(numerator, denominator);
return fraction;
}

void print()
{
if (numerator == denominator) {
System.out.println("1");
} else {
System.out.println(numerator+"/"+denominator);
    }
}

public static void main(String[] args) {
    Scanner in = new Scanner(System.in);
    Fraction fraction1 = new Fraction(in.nextInt(), in.nextInt());
    Fraction fraction2 = new Fraction(in.nextInt(),in.nextInt());
    fraction1.print();
    fraction2.print();
    fraction1.plus(fraction2).print();
    fraction1.multiply(fraction2).plus(new Fraction(5,6)).print();
    fraction1.print();
    fraction2.print();
    in.close();
    }
}
